stripe_100_node_execution_time <- stripe_100_node %>%
  select(file, executionTime, executionTimeIcycle) %>%
  pivot_longer(
    cols = c(executionTime, executionTimeIcycle),
    names_to = "Type",
    values_to = "Value"
  )

ggplot(stripe_100_node_execution_time, aes(x = Type, y = Value, fill = Type)) +
  geom_boxplot() +
  labs(title = "Execution Time Comparison 100 nodes",
       x = "Type",
       y = "Execution Time (milliseconds)") +
  theme_minimal(base_size = 16)
