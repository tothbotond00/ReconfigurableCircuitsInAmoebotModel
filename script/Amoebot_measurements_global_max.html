global_maximum_25_node_execution_time <- global_maximum_25_node %>%
  select(file, executionTime, executionTimeIcycle) %>%
  pivot_longer(
    cols = c(executionTime, executionTimeIcycle),
    names_to = "Type",
    values_to = "Value"
  )

ggplot(global_maximum_25_node_execution_time, aes(x = Type, y = Value, fill = Type)) +
  geom_boxplot() +
  labs(title = "Execution Time Comparison 25 nodes",

       y = "Execution Time (milliseconds)") +
  theme_minimal(base_size = 16)
